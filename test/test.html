First some sturdy Rust

struct Fibonacci { a: u64, b: u64 }

impl Fibonacci {
    fn new() -> Self { Fibonacci { a: 0, b: 1 } }
}

impl Iterator for Fibonacci {
    type Item = u64;
    fn next(&mut self) -> Option {
        let next = self.a + self.b;
        self.a = self.b;
        self.b = next;
        Some(self.a)
    }
}

fn main() {
    let fibs: Vec = Fibonacci::new().take(10).collect();
    println!("{:?}", fibs);
}

Then some stealthy PHP

function fibonacci(int $n): Generator {
    [$a, $b] = [0, 1];
    for ($i = 0; $i < $n; $i++) {
        yield $a;
        [$a, $b] = [$b, $a + $b];
    }
}

foreach (fibonacci(10) as $value) {
    echo $value . "\n";
}

Python (pre with class, no nested code element)

def fibonacci(n):
    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

print(fibonacci(10))
